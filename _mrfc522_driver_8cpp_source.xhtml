26 auto Mrfc522Driver::hardReset() -> void
27 {
28 // First set the resetPowerDownPin as digital input, to check the MFRC522 power down mode.
29 pinMode(pins.mfrc522.reset_pin, INPUT);
30
31 if (digitalRead(pins.mfrc522.reset_pin) == LOW)
32 { // The MFRC522 chip is in power down mode.
33 pinMode(pins.mfrc522.reset_pin, OUTPUT); // Now set the resetPowerDownPin as digital output.
34 digitalWrite(pins.mfrc522.reset_pin, LOW); // Make sure we have a clean LOW state.
35 delayMicroseconds(2); // 8.8.1 Reset timing requirements says about 100ns. Let us be generous: 2μsl
36 digitalWrite(pins.mfrc522.reset_pin, HIGH); // Exit power down mode. This triggers a hard reset.
37 // Section 8.8.2 in the datasheet says the oscillator start-up time is the start up time of the crystal + 37,74μs. Let us be generous: 50ms.
38 delay(50);
39 }
40 }
