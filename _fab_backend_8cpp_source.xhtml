61 auto FabBackend::publishWithReply(const MQTTInterface::Query &query) -> PublishResult
62 {
63 auto try_cpt = 0;
64 auto published = false;
65
66 while (try_cpt < conf::mqtt::MAX_TRIES)
67 {
68 try_cpt++;
69
70 if (!published)
71 {
72 if (publish(query) == PublishResult::PublishedWithoutAnswer)
73 {
74 published = true;
75 }
76 else
77 {
78 ESP_LOGE(TAG, "MQTT Client: failure to send query %s", query.payload().data());
79 Tasks::delay(conf::mqtt::TIMEOUT_REPLY_SERVER);
80 }
81 }
82
83 if (waitForAnswer(conf::mqtt::TIMEOUT_REPLY_SERVER))
84 {
85 ESP_LOGD(TAG, "MQTT Client: received answer: %s", last_reply.data());
86 last_unresponsive = std::nullopt;
87 return PublishResult::PublishedWithAnswer;
88 }
89
90 ESP_LOGW(TAG, "MQTT Client: no answer received, retrying %d/%d", try_cpt, conf::mqtt::MAX_TRIES);
91 Tasks::delay(conf::mqtt::TIMEOUT_REPLY_SERVER);
92 }
93
94 ESP_LOGE(TAG, "MQTT Client: failure to send query %s", query.payload().data());
95
96 // Do not send twice if response did not arrive
97 if (query.buffered() && !published)
98 {
99 const auto &msg = BufferedMsg{query.payload(), topic, query.waitForReply()};
100 buffer.push_back(msg);
101 }
102
103 if (published)
104 {
105 last_unresponsive = Tasks::arduinoNow();
106 return PublishResult::PublishedWithoutAnswer;
107 }
108
109 return PublishResult::ErrorNotPublished;
110 }

206 bool FabBackend::waitForAnswer(std::chrono::milliseconds max_duration)
207 {
208 const auto start_time = Tasks::arduinoNow();
209 const auto DELAY_MS = 25ms;
210 do
211 {
212 if (answer_pending)
213 {
214 client.loop();
215 if (!client.connected())
216 {
217 ESP_LOGW(TAG, "MQTT Client: connection lost while waiting for answer");
218 connect();
219 }
220 Tasks::delay(DELAY_MS);
221 }
222 else
223 {
224 return true;
225 }
226 } while (Tasks::arduinoNow() < (start_time + max_duration));
227
228 ESP_LOGE(TAG, "Failure, no answer from MQTT server (timeout:%" PRId64 " ms)", max_duration.count());

448 std::unique_ptr<RespT> FabBackend::processQuery(QueryArgs &&...args)
449 {
450 static_assert(std::is_base_of<MQTTInterface::Query, QueryT>::value, "QueryT must inherit from Query");
451 static_assert(std::is_base_of<MQTTInterface::Response, RespT>::value, "RespT must inherit from Response");
452 QueryT query{args...};
453
454 auto nb_tries = 0;
455 while (isOnline() && hasBufferedMsg() && !transmitBuffer() && nb_tries < 3)
456 {
457 // To preserve chronological order, we cannot send new messages until the old ones have been sent.
458 ESP_LOGW(TAG, "Online with pending messages that could not be transmitted, retrying...");
459
460 if (!shouldFailFast())
461 {
462 connect();
463 }
464 else
465 {
466 ESP_LOGW(TAG, "processQuery: failing fast due to unresponsive backend.");
467 break;
468 }
469
470 Tasks::delay(250ms);
471 nb_tries++;
472 }
473
474 if (isOnline() && !hasBufferedMsg())
475 {
476 if (publishWithReply(query) == PublishResult::PublishedWithAnswer)
477 {
478 // Deserialize the JSON document
479 const auto payload = last_reply.c_str();
480 if (DeserializationError error = deserializeJson(doc, payload))
481 {
482 ESP_LOGE(TAG, "Failed to parse json: %s (%s)", payload, error.c_str());
483 return std::make_unique<RespT>(false);
484 }
485 return RespT::fromJson(doc);
486 }
487 else
488 {
489 ESP_LOGE(TAG, "Failed to publish query %s", query.payload().data());
490 this->disconnect();
491 }
492 }
493
494 if (query.buffered())
495 {
496 const auto &msg = BufferedMsg{query.payload(), topic, query.waitForReply()};
497 buffer.push_back(msg);
498 }
499
500 return std::make_unique<RespT>(false);

511 bool FabBackend::processQuery(QueryArgs &&...args)
512 {
513 static_assert(std::is_base_of<MQTTInterface::Query, QueryT>::value, "QueryT must inherit from Query");
514 QueryT query{args...};
515
516 auto nb_tries = 0;
517 while (isOnline() && hasBufferedMsg() && !transmitBuffer() && nb_tries < 3)
518 {
519 // To preserve chronological order, we cannot send new messages until the old ones have been sent.
520 ESP_LOGW(TAG, "Online with pending messages that could not be transmitted, retrying...");
521
522 if (!isOnline())
523 {

525 {
526 connect();

528 else
529 {
530 ESP_LOGW(TAG, "processQuery: failing fast due to unresponsive backend.");
531 break;
532 }
533 }
534
535 Tasks::delay(250ms);
536 nb_tries++;
537 }
538
539 if (isOnline() && !hasBufferedMsg())
540 {
541 if (publish(query) == PublishResult::PublishedWithoutAnswer)
542 {
543 return true;

545
546 ESP_LOGW(TAG, "Failed to publish query %s", query.payload().data());
547 this->disconnect();
548 }
549
550 if (query.buffered())
551 {
552 const auto &msg = BufferedMsg{query.payload(), topic, query.waitForReply()};
553 buffer.push_back(msg);
554 }
555 return false;

650 [[nodiscard]] auto FabBackend::transmitBuffer() -> bool
651 {
652 while (hasBufferedMsg() && isOnline())
653 {
654 ESP_LOGD(TAG, "Retransmitting buffered messages...");
655 const auto &msg = buffer.getMessage();
656 const BufferedQuery bq{msg.mqtt_message, msg.mqtt_topic, msg.wait_for_answer};
657 if (bq.waitForReply())
658 {
659 if (auto result = publishWithReply(bq); result != PublishResult::PublishedWithAnswer)
660 {
661 ESP_LOGW(TAG, "Retransmitting buffered message failed!");
662
663 // If it has been published but not answered, do not try again
664 if (result == PublishResult::PublishedWithoutAnswer)
665 {
666 const auto &bmsg = BufferedMsg{msg.mqtt_message, msg.mqtt_topic, msg.wait_for_answer};
667 buffer.push_front(bmsg);
668 }
669 break;
670 }
671 }
672 else
673 {
674 if (publish(bq) == PublishResult::ErrorNotPublished)
675 {
676 // Will try again
677 ESP_LOGW(TAG, "Retransmitting buffered message failed!");
678 const auto &bmsg = BufferedMsg{msg.mqtt_message, msg.mqtt_topic, msg.wait_for_answer};
679 buffer.push_front(bmsg);
680 break;
681 }
682 }
683 }
684 last_reply.clear();
685
686 ESP_LOGW(TAG, "Retransmittion completed, remaining messages=%d", buffer.count());
687 return !hasBufferedMsg();
688 }
